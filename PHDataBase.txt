create table info_patient(
    gender varchar(10);
    age int not null;
    name varchar not null;
    password_hash text not null;
    email varchar(255) unique not null;
    height decimal(5,2);
    weight decimal(5,2);
    allergies text;
    phone_number varchar(20) not null;
);

create table info_doctor(
    doctor_specialty text not null;
    work_city text;
    work_place text;
    uin varchar(10);
)
